export const InputTemplate = (args) => (
	<Input placeholder={args.placeholder} name="username" type={InputType.TEXT}>
		<Input.Label
			type={args.labelType}
			htmlFor="username"
			color={args.labelColor}
			weight={args.labelWeight}
			variant={args.labelVariant}
			size={args.labelSize}
			letterCase={args.labelLetterCase}

		<Input.Icon
